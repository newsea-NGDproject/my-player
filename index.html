// タップの間隔を精査して補正する関数 (ロジック部分の抜粋) function getCorrectedBPM(taps) { let intervals = []; for (let i = 1; i < taps.length; i++) { intervals.push(taps[i] - taps[i-1]); } // 1. まずはおおよその「1拍」の長さを決める (中央値) const sorted = [...intervals].sort(); const medianInterval = sorted[Math.floor(sorted.length / 2)]; let totalBeats = 0; let totalTime = 0; // 2. 各間隔が「何拍分」に相当するか判定して足し合わせる intervals.forEach(interval => { // 間隔が標準の何倍か？ (例: 1.9倍なら 2拍、3.1倍なら 3拍と判定) const beatsInInterval = Math.round(interval / medianInterval); // 異常に離れすぎ（0.5倍未満や4倍以上）でなければ採用 if (beatsInInterval >= 1 && beatsInInterval <= 3) { totalBeats += beatsInInterval; totalTime += interval; } }); return (totalBeats * 60) / totalTime; }
